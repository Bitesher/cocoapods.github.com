PathList

The PathList class is designed to perform multiple glob matches against a given directory. Basically, it generates a list of all the children paths and matches the globs patterns against them, resulting in just one access to the file system.

45 def read_file_system
46   root_length = root.to_s.length+1
47   paths = Dir.glob(root + "**/*", File::FNM_DOTMATCH)
48   paths = paths.map { |p| p[root_length..-1] }
49   paths = paths.reject { |p| p == '.' || p == '..' }
50   dirs_entries = paths.select { |path| path.end_with?('/.', '/..') }
51   @files = paths - dirs_entries
52   @dirs  = dirs_entries.map { |d| d.gsub(/\/\.\.?$/,'') }.uniq
53 end

73 def relative_glob(patterns, dir_pattern = nil, exclude_patterns = nil)
74   return [] if patterns.empty?
75   patterns = [ patterns ] if patterns.is_a? String
76 
77   list = patterns.map do |pattern|
78     if pattern.is_a?(String)
79       pattern += '/' + dir_pattern if directory?(pattern) && dir_pattern
80       expanded_patterns = dir_glob_equivalent_patterns(pattern)
81       files.select do |path|
82         expanded_patterns.any? do |p|
83           File.fnmatch(p, path, File::FNM_CASEFOLD | File::FNM_PATHNAME)
84         end
85       end
86     else
87       files.select { |path| path.match(pattern) }
88     end
89   end.flatten
90 
91   list -= relative_glob(exclude_patterns) if exclude_patterns
92 
93   list.map { |path| Pathname.new(path) }
94 end

126 def dir_glob_equivalent_patterns(pattern)
127   pattern.gsub!('/**/', '{/**/,/}')
128   values_by_set = {}
129   pattern.scan(/\{[^}]*\}/) do |set|
130     values = set.gsub(/[{}]/, '').split(',')
131     values_by_set[set] = values
132   end
133 
134   if values_by_set.empty?
135     [ pattern ]
136   else
137     patterns = [ pattern ]
138     values_by_set.each do |set, values|
139       patterns = patterns.map do |pattern|
140         values.map do |value|
141           pattern.gsub(set, value)
142         end
143       end.flatten
144     end

146   end
