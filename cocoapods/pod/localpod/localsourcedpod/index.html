LocalSourcedPod

A LocalSourcedPod is a LocalPod that interacts with the files of a folder controlled by the users. As such this class does not alter in any way the contents of the folder.

Inherits from:

Pod::LocalPod

606 def expanded_paths(patterns, dir_pattern = nil, exclude_patterns = nil)
607   unless exists?
608     raise Informative, "[Local Pod] Attempt to resolve paths for nonexistent pod.\n" \
609                        "\tSpecifications: #{@specifications.inspect}\n" \
610                        "\t      Patterns: #{patterns.inspect}"
611   end
612 
613   # Noticeable impact on performance
614   return [] if patterns.empty?
615 
616   patterns = [ patterns ] if patterns.is_a?(String)
617   file_lists = patterns.select { |p| p.is_a?(FileList) }
618   glob_patterns = patterns - file_lists
619 
620   result = []
621 
622   result << path_list.glob(glob_patterns, dir_pattern, exclude_patterns)
623

627   end
628
