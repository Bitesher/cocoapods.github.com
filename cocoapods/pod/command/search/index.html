def run
  sets = Source.search_by_name(@query.strip, @full_text_search)
  sets.each do |set|
    begin
      UI.pod(set, (@stats ? :stats : :normal))
    rescue DSLError => e
      UI.warn "Skipping `#{set.name}` because the podspec contains errors."
    end
  end
end

def self.report_error(error)
  if error.is_a?(Interrupt)
    puts "[!] Cancelled".red
    Config.instance.verbose? ? raise : exit(1)
  else
    puts ErrorReport.report(error)
    exit 1
  end
end
