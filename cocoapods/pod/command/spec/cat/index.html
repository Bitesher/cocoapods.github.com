230 def suggested_ref_and_version(repo)
231   tags = Octokit.tags(:username => repo['owner']['login'], :repo => repo['name']).map {|tag| tag["name"]}
232   versions_tags = {}
233   tags.each do |tag|
234     clean_tag = tag.gsub(/^v(er)? ?/,'')
235     versions_tags[Gem::Version.new(clean_tag)] = tag if Gem::Version.correct?(clean_tag)
236   end
237   version = versions_tags.keys.sort.last || '0.0.1'
238   data = {:version => version}
239   if version == '0.0.1'
240     branches        = Octokit.branches(:username => repo['owner']['login'], :repo => repo['name'])
241     master_name     = repo['master_branch'] || 'master'
242     master          = branches.select {|branch| branch['name'] == master_name }.first
243     data[:ref_type] = ':commit'
244     data[:ref]      = master['commit']['sha']
245   else
246     data[:ref_type] = ':tag'
247     data[:ref]      = versions_tags[version]
248   end
249   data
250 end
