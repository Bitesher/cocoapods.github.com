427 def suggested_ref_and_version(repo)
428   tags = Octokit.tags(:username => repo['owner']['login'], :repo => repo['name']).map {|tag| tag["name"]}
429   versions_tags = {}
430   tags.each do |tag|
431     clean_tag = tag.gsub(/^v(er)? ?/,'')
432     versions_tags[Gem::Version.new(clean_tag)] = tag if Gem::Version.correct?(clean_tag)
433   end
434   version = versions_tags.keys.sort.last || '0.0.1'
435   data = {:version => version}
436   if version == '0.0.1'
437     branches        = Octokit.branches(:username => repo['owner']['login'], :repo => repo['name'])
438     master_name     = repo['master_branch'] || 'master'
439     master          = branches.select {|branch| branch['name'] == master_name }.first
440     data[:ref_type] = ':commit'
441     data[:ref]      = master['commit']['sha']
442   else
443     data[:ref_type] = ':tag'
444     data[:ref]      = versions_tags[version]
445   end
446   data
447 end
