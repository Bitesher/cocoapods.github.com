TargetHeader

Generates a header which allows to inspect at compile time the installed pods and the installed specifications of a pod.

Example output:

#define __COCOA_PODS

#define __POD_AFIncrementaStore
#define __POD_AFNetworking
#define __POD_libextobjc_EXTConcreteProtocol
#define __POD_libextobjc_EXTKeyPathCoding
#define __POD_libextobjc_EXTScope

Example usage:

#ifdef __COCOA_PODS
  #ifdef __POD__AFNetworking
    #import "MYLib+AFNetworking.h"
  #endif
#else
  // Non CocoaPods code
#endif
