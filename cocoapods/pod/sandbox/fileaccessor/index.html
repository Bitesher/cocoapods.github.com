188 def expanded_paths(patterns, options = {})
189   return [] if patterns.empty?
190 
191   file_lists = patterns.select { |p| p.is_a?(FileList) }
192   glob_patterns = patterns - file_lists
193 
194   result = []
195   result << path_list.glob(glob_patterns, options)
196   result << file_lists.map do |file_list|
197     file_list.prepend_patterns(path_list.root)
198     file_list.glob
199   end
200 
201   unless file_lists.empty?
202     # TODO Restore warning in 0.17 proper
203     # UI.warn "[#{spec_consumer.spec.name}] The usage of Rake FileList is deprecated. Use `exclude_files`."
204   end
205 
206   result.flatten.compact.uniq
207 end
