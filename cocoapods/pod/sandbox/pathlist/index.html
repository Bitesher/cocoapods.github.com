PathList

The PathList class is designed to perform multiple glob matches against a given directory. Basically, it generates a list of all the children paths and matches the globs patterns against them, resulting in just one access to the file system.

45 def read_file_system
46   unless root.exist?
47     raise Informative, "Attempt to read non existent folder `#{root}`."
48   end
49   root_length  = root.to_s.length+1
50   escaped_root = escape_path_for_glob(root)
51   paths  = Dir.glob(escaped_root + "**/*", File::FNM_DOTMATCH)
52   absolute_dirs  = paths.select { |path| File.directory?(path) }
53   relative_dirs  = absolute_dirs.map  { |p| p[root_length..-1] }
54   absolute_paths = paths.reject { |p| p == "#{root}/." || p == "#{root}/.." }
55   relative_paths = absolute_paths.map { |p| p[root_length..-1] }
56   @files = relative_paths - relative_dirs
57   @dirs  = relative_dirs.map { |d| d.gsub(/\/\.\.?$/,'') }.reject { |d| d == '.' || d == '..' } .uniq
58 end

 84 def relative_glob(patterns, options = {})
 85   return [] if patterns.empty?
 86 
 87   dir_pattern = options[:dir_pattern]
 88   exclude_patterns = options[:exclude_patterns]
 89   include_dirs = options[:include_dirs]
 90 
 91   if include_dirs
 92     full_list = files + dirs
 93   else
 94     full_list = files
 95   end
 96 
 97   list = Array(patterns).map do |pattern|
 98     if pattern.is_a?(String)
 99       if directory?(pattern) && dir_pattern
100         pattern += '/' unless pattern.end_with?('/')
101         pattern +=  dir_pattern
102       end
103       expanded_patterns = dir_glob_equivalent_patterns(pattern)
104       full_list.select do |path|
105         expanded_patterns.any? do |p|
106           File.fnmatch(p, path, File::FNM_CASEFOLD | File::FNM_PATHNAME)
107         end
108       end
109     else
110       full_list.select { |path| path.match(pattern) }
111     end
112   end.flatten
113 
114   list = list.map { |path| Pathname.new(path) }
115   if exclude_patterns
116     exclude_options = { :dir_pattern => '**/*', :include_dirs => include_dirs }
117     list -= relative_glob(exclude_patterns, exclude_options)
118   end
119   list
120 end

158 def dir_glob_equivalent_patterns(pattern)
159   pattern = pattern.gsub('/**/', '{/**/,/}')
160   values_by_set = {}
161   pattern.scan(/\{[^}]*\}/) do |set|
162     values = set.gsub(/[{}]/, '').split(',')
163     values_by_set[set] = values
164   end
165 
166   if values_by_set.empty?
167     [ pattern ]
168   else
169     patterns = [ pattern ]
170     values_by_set.each do |set, values|
171       patterns = patterns.map do |old_pattern|
172         values.map do |value|
173           old_pattern.gsub(set, value)
174         end
175       end.flatten
176     end
177     patterns
178   end
