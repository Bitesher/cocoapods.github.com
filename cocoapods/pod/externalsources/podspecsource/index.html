133 def store_podspec(sandbox, podspec)
134   output_path = sandbox.root + "Local Podspecs/#{name}.podspec"
135   output_path.dirname.mkpath
136   if podspec.is_a?(String)
137     unless podspec.include?('Spec.new')
138       raise Informative, "The `#{name}.podspec` from `#{description}` appears to be invalid."
139     end
140     output_path.open('w') { |f| f.puts(podspec) }
141   else
142     unless podspec.exist?
143       raise Informative, "No podspec found for `#{name}` in #{description}"
144     end
145     FileUtils.copy(podspec, output_path)
146   end
147 end
