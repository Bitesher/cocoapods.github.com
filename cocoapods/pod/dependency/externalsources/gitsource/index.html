180 def store_podspec(sandbox, podspec)
181   output_path = sandbox.root + "Local Podspecs/#{name}.podspec"
182   output_path.dirname.mkpath
183   if podspec.is_a?(String)
184     raise Informative, "No podspec found for `#{name}' in #{description}" unless podspec.include?('Spec.new')
185     output_path.open('w') { |f| f.puts(podspec) }
