 95 def add_target
 96   name     = library.label
 97   platform = library.platform.name
 98   settings = build_settings_for_platform(library.platform)
 99   settings['GCC_PREFIX_HEADER'] = library.prefix_header_name
100   if target_definition.inhibit_all_warnings?
101    settings['GCC_WARN_INHIBIT_ALL_WARNINGS'] = 'YES'
102   end
103   Generator::XCConfig.pods_project_settings.each do |key, value|
104     settings[key] = value
105   end
106 
107   @target = project.new_target(:static_library, name, platform)
108   @target.build_settings('Debug').merge!(settings)
109   @target.build_settings('Release').merge!(settings)
110 
111   library.user_build_configurations.each do |name, type|
112     unless @target.build_configurations.map(&:name).include?(name)
113       build_config = project.new(Xcodeproj::Project::XCBuildConfiguration)
114       build_config.name = name
115       settings = @target.build_settings(type.to_s.capitalize)
116       build_config.build_settings = settings
117       target.build_configurations << build_config
118       project.build_configurations << build_config
119     end
120   end
121 
122   @target.build_configurations.each do |c|
123     c.base_configuration_reference = xcconfig_file_ref
124   end
125 
126   library.target = @target
127 end

242 def build_settings_for_platform(platform)
243   settings = {}
244   settings['ARCHS'] = "armv6 armv7" if platform.requires_legacy_ios_archs?
245 
246   if dt = platform.deployment_target
247     if platform == :ios
248       settings['IPHONEOS_DEPLOYMENT_TARGET'] = dt.to_s
249     elsif platform == :osx
250       settings['MACOSX_DEPLOYMENT_TARGET'] = dt.to_s
251     else
252       raise Informative, "Unknown platform #{platform}"
253     end
254   end
255   settings
256 end
