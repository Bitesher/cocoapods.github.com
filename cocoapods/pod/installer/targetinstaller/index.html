106 def add_target
107   name = library.label
108   platform = library.platform.name
109   deployment_target = library.platform.deployment_target.to_s
110   @target = project.new_target(:static_library, name, platform, deployment_target)
111 
112   settings = {}
113   if library.platform.requires_legacy_ios_archs?
114     settings['ARCHS'] = "armv6 armv7"
115   end
116   if target_definition.inhibit_all_warnings?
117     settings['GCC_WARN_INHIBIT_ALL_WARNINGS'] = 'YES'
118   end
119 
120   @target.build_settings('Debug').merge!(settings)
121   @target.build_settings('Release').merge!(settings)
122 
123   library.user_build_configurations.each do |name, type|
124     unless @target.build_configurations.map(&:name).include?(name)
125       build_config = project.new(Xcodeproj::Project::XCBuildConfiguration)
126       build_config.name = name
127       settings = @target.build_settings(type.to_s.capitalize)
128       build_config.build_settings = settings
129       target.build_configurations << build_config
130       project.build_configurations << build_config
131     end
132   end
133 
134 library.target = @target
135 end
