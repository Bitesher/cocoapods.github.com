312 def generate_sandbox_state
313   sandbox_lockfile = sandbox.manifest  || lockfile
314   sandbox_state = SpecsState.new
315 
316   UI.section "Comparing resolved specification to the sandbox manifest:" do
317     resolved_subspecs_names = specifications.group_by { |s| s.root.name }
318     resolved_names          = resolved_subspecs_names.keys
319 
320     if sandbox_lockfile
321       sandbox_subspecs_names = sandbox_lockfile.pod_names.group_by { |name| Specification.root_name(name) }
322       sandbox_names = sandbox_subspecs_names.keys
323       all_names     = (resolved_names + sandbox_names).uniq
324       root_specs    = specifications.map(&:root).uniq
325 
326       is_changed = lambda do |name|
327         spec = root_specs.find { |spec| spec.name == name }
328         spec.version != sandbox_lockfile.version(name) \
329           || spec.checksum != sandbox_lockfile.checksum(name) \
330           || resolved_subspecs_names[name] =! sandbox_subspecs_names[name] \
331       end
332 
333       all_names.each do |name|
334         state = case
335                 when resolved_names.include?(name) && !sandbox_names.include?(name) then :added
336                 when !resolved_names.include?(name) && sandbox_names.include?(name) then :deleted
337                 when is_changed.call(name) then :changed
338                 else :unchanged
339                 end
340         sandbox_state.add_name(name, state)
341       end
342 
343     else
344       sandbox_state.added.concat(resolved_names)
345     end
346     sandbox_state.print
347   end
348   sandbox_state
349 end

375 def compute_user_project_path(target_definition)
376   if target_definition.user_project_path
377     path = config.project_root + target_definition.user_project_path
378     path = "#{path}.xcodeproj" unless File.extname(path) == '.xcodeproj'

394   end

412 def compute_user_project_targets(target_definition, user_project)
413   if link_with = target_definition.link_with
414     targets = user_project.targets.select { |t| link_with.include? t.name }
415     raise Informative, "Unable to find the targets named `#{link_with.to_sentence}` to link with target definition `#{target_definition.name}`" if targets.empty?
416   elsif target_definition.name != :default
417     target = user_project.targets.find { |t| t.name == target_definition.name.to_s }
418     targets = [ target ].compact
419     raise Informative, "Unable to find a target named `#{target_definition.name.to_s}`" if targets.empty?
420   else

422     raise Informative, "Unable to find a target" if targets.empty?
