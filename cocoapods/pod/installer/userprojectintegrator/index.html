 82 def create_workspace
 83   all_projects = user_project_paths.sort.push(sandbox.project_path).uniq
 84   projpaths = all_projects.map do |path|
 85     path.relative_path_from(workspace_path.dirname).to_s
 86   end
 87 
 88   if workspace_path.exist?
 89     workspace = Xcodeproj::Workspace.new_from_xcworkspace(workspace_path)
 90     new_projpaths = projpaths - workspace.projpaths
 91     unless new_projpaths.empty?
 92       workspace.projpaths.concat(new_projpaths)
 93       workspace.save_as(workspace_path)
 94     end
 95 
 96   else
 97     UI.notice "From now on use `#{workspace_path.basename}`."
 98     workspace = Xcodeproj::Workspace.new(*projpaths)
 99     workspace.save_as(workspace_path)
100   end
101 end
