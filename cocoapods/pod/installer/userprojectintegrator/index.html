64 def create_workspace
65   workspace = Xcodeproj::Workspace.new_from_xcworkspace(workspace_path)
66   [pods_project.path, *user_project_paths].each do |project_path|
67     path = project_path.relative_path_from(workspace_path.dirname).to_s
68     workspace << path unless workspace.include?(path)
69   end
70   unless workspace_path.exist?
71     UI.notice "From now on use `#{workspace_path.basename}'."
72   end
73   workspace.save_as(workspace_path)
