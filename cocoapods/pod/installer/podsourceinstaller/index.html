249 def clean_paths
250   cached_used = used_files
251   glob_options = File::FNM_DOTMATCH | File::FNM_CASEFOLD
252   files = Pathname.glob(root + "**/*", glob_options).map(&:to_s)
253 
254   files.reject! do |candidate|
255     candidate = candidate.downcase
256     candidate.end_with?('.', '..') || cached_used.any? do |path|
257       path = path.downcase
258       path.include?(candidate) || candidate.include?(path)
259     end
260   end
261   files
262 end
