167 def find_dependency_specs(dependent_spec, dependencies, target_definition)
168   dependencies.each do |dependency|
169     locked_dep = locked_dependencies.find { |ld| ld.name == dependency.name }
170     dependency = locked_dep if locked_dep
171 
172     UI.message("- #{dependency}", '', 2) do
173       set = find_cached_set(dependency)
174       set.required_by(dependency, dependent_spec.to_s)
175 
176       unless @loaded_specs.include?(dependency.name)
177         spec = set.specification.subspec_by_name(dependency.name)
178         @loaded_specs << spec.name
179         cached_specs[spec.name] = spec
180         validate_platform(spec, target_definition)
181         spec.version.head = dependency.head?
182 
183         spec_dependencies = spec.all_dependencies(target_definition.platform)
184         find_dependency_specs(spec, spec_dependencies, target_definition)
185       end
186     end
187   end
188 end

223 def set_from_external_source(dependency)
224   source = ExternalSources.from_dependency(dependency)
225   if allow_pre_downloads?
226     if update_external_specs?
