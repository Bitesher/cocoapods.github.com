128 def update(source_name = nil, show_output = false)
129   if source_name
130     specified_source = aggregate.all.find { |s| s.name == source_name }
131     raise Informative, "Unable to find the `#{source_name}` repo."    unless specified_source
132     raise Informative, "The `#{source_name}` repo is not a git repo." unless git_repo?(specified_source.repo)
133     sources = [specified_source]
134   else
135     sources = aggregate.all.select { |source| git_repo?(source.repo) }
136   end
137 
138   sources.each do |source|
139     UI.section "Updating spec repo `#{source.name}`" do
140       Dir.chdir(source.repo) do
141         output = git!("pull")
142         UI.puts output if show_output && !config.verbose?
143       end
144       check_version_information(source.repo)
145     end
146   end
147 end

172 def check_version_information(dir)
173   versions = version_information(dir)
174   unless repo_compatible?(dir)
175     min, max = versions['min'], versions['max']
176     version_msg = ( min == max ) ? min : "#{min} - #{max}"

184   end

195 def repo_compatible?(dir)
196   versions = version_information(dir)
197 
198   min, max = versions['min'], versions['max']
199   bin_version  = Gem::Version.new(Pod::VERSION)
200   supports_min = !min || bin_version >= Gem::Version.new(min)
201   supports_max = !max || bin_version <= Gem::Version.new(max)
202   supports_min && supports_max
203 end
