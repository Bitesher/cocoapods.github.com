def message
  message = [
    '',
    @command_class.banner.gsub(/\$ pod (.*)/, '$ pod \1'.green),
    '',
    'Options:',
    '',
    options,
    "\n",
  ].join("\n")
  message << "[!] Unrecognized command: `#{@unrecognized_command}'\n".red if @unrecognized_command
  message << "[!] Unrecognized argument#{@argv.count > 1 ? 's' : ''}: `#{@argv.join(' - ')}'\n".red unless @argv.empty?
  message
end
