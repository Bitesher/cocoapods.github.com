def attribute(name, options = {})
  attr = Attribute.new(name, options)
  @attributes ||= []
  @attributes << attr
  unless attr.skip_definitions?
    define_attr_reader(attr)
    define_attr_writer(attr)
    define_attr_writer_singular_form(attr)
  end
end
