def to_s
  r = "[#{type.to_s.upcase}] #{message}"
  if platforms != Specification::PLATFORMS
    platforms_names = platforms.uniq.map { |p| Platform.string_name(p) }
    r << " [#{platforms_names * ' - '}]" unless platforms.empty?
  end
  r
end
