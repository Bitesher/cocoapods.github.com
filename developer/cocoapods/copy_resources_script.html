def save_as(pathname)
  pathname.open('w') do |script|
    script.puts CONTENT
    @resources.each do |resource|
      script.puts "install_resource '#{resource}'"
    end
  end
  # TODO use File api
  system("chmod +x '#{pathname}'")
end
