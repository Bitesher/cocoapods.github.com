182 def prune_cache
183   return unless cache_root && class_cache_dir.exist?
184   Dir.chdir(class_cache_dir) do
185     repos = Pathname.new(class_cache_dir).children.select { |c| c.directory? }.sort_by(&:ctime)
186     while caches_size >= max_cache_size && !repos.empty?
