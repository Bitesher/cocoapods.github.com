152 def prune_cache
153   return unless cache_root && class_cache_dir.exist?
154   Dir.chdir(class_cache_dir) do
155     repos = Pathname.new(class_cache_dir).children.select { |c| c.directory? }.sort_by(&:ctime)
156     while caches_size >= max_cache_size && !repos.empty?
157       dir = repos.shift
158       # UI.message "#{'->'.yellow} Removing #{name} cache for `#{cache_origin_url(dir)}'"
159       dir.rmtree
160     end
161   end
162 end
