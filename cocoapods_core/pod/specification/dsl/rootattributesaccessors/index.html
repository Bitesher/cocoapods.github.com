47 def authors
48   authors = attributes_hash["authors"]
49   if authors.is_a?(Hash)
50     authors
51   elsif authors.is_a?(Array)

63   end

139 def convert_keys_to_symbol(value)
140   return unless value
141   result = {}
142   value.each do |key, subvalue|
143     subvalue = convert_keys_to_symbol(subvalue) if subvalue.is_a?(Hash)
144     result[key.to_sym] = subvalue
145   end
146   result
147 end
