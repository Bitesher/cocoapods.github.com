Acceptor

Checks whether a podspec can be accepted by a source. The check takes into account the introduction of 0.0.1 version if there are already tagged ones or whether there is change in the source.

63 def check_spec_source_change(spec, errors)
64   return unless spec
65   return if spec.source[:http]
66   return unless reference_spec(spec)
67   keys = Spec::DSL::SOURCE_KEYS.keys
68   source = spec.source.values_at(*keys).compact.first
69   old_source = reference_spec(spec).source.values_at(*keys).compact.first
70   unless source == old_source
71     errors << "The source of the spec doesn't match with the recorded ones." \
72       "Source: `#{source}`. Previous: `#{old_source}`.\n " \
73       "Please contact the specs repo maintainers if the library changed " \
74       "location."
75   end
76 end

84 def check_if_untagged_version_is_acceptable(spec, previous_spec, errors)
85   return if !spec.source[:git] || spec.source[:tag]
86   return unless related_specifications(spec)
87   return if previous_spec
88   has_tagged_spec = related_specifications(spec).any? { |s| s.version != '0.0.1' }
89   if has_tagged_spec
90     errors << "There is already at least one versioned specification so " \
91       "untagged versions cannot be accepted."
92   end
93 end

136 def related_specifications(spec)
137   versions = source.versions(spec.name)
138   return unless versions
139   specs = versions.sort.map { |v| source.specification(spec.name, v) }
140   specs.delete(spec)
141   specs
142 end
