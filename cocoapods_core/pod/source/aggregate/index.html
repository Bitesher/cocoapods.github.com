256 def search_by_name(query, full_text_search = false)
257   pods_by_source = {}
258   result = []
259   all.each { |s| pods_by_source[s] = s.search_by_name(query, full_text_search).map(&:name) }
260   root_spec_names = pods_by_source.values.flatten.uniq
261   root_spec_names.each do |pod|
262     sources = []
263     pods_by_source.each{ |source, pods| sources << source if pods.include?(pod) }
264     result << Specification::Set.new(pod, sources)
265   end
266   if result.empty?
267     extra = ", author, summary, or description" if full_text_search
268     raise(Informative, "Unable to find a pod with name" \
269           "#{extra} matching `#{query}'")
270   end
271   result
272 end
