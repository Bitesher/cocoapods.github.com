YAMLConverter

Converts objects to their YAML representation.

This class was created for the need having control on how the YAML is representation is generated. In details it provides:

  • sorting for hashes in ruby 1.8.x
  • ability to hint the sorting of the keys of a dictionary when converting it. In this case the keys are also separated by an additional new line feed for readability.

61 def process_according_to_class(value, hash_keys_hint = nil)
62   case value
63   when String     then value
64   when Symbol     then ":#{value}"
65   when TrueClass  then 'true'
66   when FalseClass then 'false'
67   when Array      then process_array(value)
68   when Hash       then process_hash(value, hash_keys_hint)
69   else
70     raise StandardError, "Unsupported class for YAML conversion #{value.class}"
71   end
72 end

104 def process_hash(hash, hash_keys_hint = nil, line_separator = "\n")
105   keys = sorted_array_with_hint(hash.keys, hash_keys_hint)
106   key_lines = []
107   keys.each do |key|
108     key_value = hash[key]
109     processed = process_according_to_class(key_value)
110     processed_key = process_according_to_class(key)
111     case key_value
112     when Array, Hash
113       key_partial_yaml = processed.lines.map { |line| "  #{line}" } * ""
114       key_lines << "#{processed_key}:\n#{key_partial_yaml}"
115     else
116       key_lines << "#{processed_key}: #{processed}"
117     end
118   end
119   key_lines * line_separator
120 end

180 def sorting_string(value)
181   return "" unless value
182   case value
183   when String then value.downcase
184   when Symbol then sorting_string(value.to_s)
185   when Array  then sorting_string(value.first)
186   when Hash   then value.keys.map { |key| key.to_s.downcase }.sort.first
187   end
188 end
