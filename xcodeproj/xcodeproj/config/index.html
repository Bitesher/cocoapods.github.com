 95 def to_hash
 96   hash = @attributes.dup
 97   flags = hash['OTHER_LDFLAGS'] || ''
 98   flags = flags.dup.strip
 99   flags << libraries.to_a.sort.reduce('')  {| memo, l | memo << " -l#{l}" }
100   flags << frameworks.to_a.sort.reduce('') {| memo, f | memo << " -framework #{f}" }
101   flags << weak_frameworks.to_a.sort.reduce('') {| memo, f | memo << " -weak_framework #{f}" }
102   hash['OTHER_LDFLAGS'] = flags.strip
103   hash.delete('OTHER_LDFLAGS') if flags.strip.empty?
104   hash
105 end

209 def hash_from_file_content(string)
210   hash = {}
211   string.split("\n").each do |line|
212     uncommented_line = strip_comment(line)
213     if include = extract_include(uncommented_line)
214       @includes.push include
215     else
216       key, value = extract_key_value(uncommented_line)
217       hash[key] = value if key
218     end
219   end
220   hash
221 end
